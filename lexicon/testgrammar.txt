PROCEDURE                   -> content:STATEMENT*;

STATEMENT                   -> type:(ASSIGN_STATEMENT | PRINT_STATEMENT | OUTPUT_STATEMENT | IF_STATEMENT | WHILE_STATEMENT);
ASSIGN_STATEMENT            -> destination:VARIABLE . operation:(Arithmetic)? . Assign . source:EXPRESSION . StatementDelimiter;
PRINT_STATEMENT             -> string:String . StatementDelimiter;
OUTPUT_STATEMENT            -> OutputDelimiter . interior:EXPRESSION . OutputDelimiter . StatementDelimiter;
IF_STATEMENT                -> KeywordIf . conditionalBlock:CONDITIONAL_BLOCK;
WHILE_STATEMENT             -> KeywordWhile . conditionalBlock:CONDITIONAL_BLOCK;
CONDITIONAL_BLOCK           -> condition:EXPRESSION . block:BLOCK;
BLOCK                       -> OpenBlock . procedure:PROCEDURE . CloseBlock;

EXPRESSION                  -> type:(OPERATION_EXPRESSION | GROUP_EXPRESSION);
GROUP_EXPRESSION            -> OpenExpression . interior:(EXPRESSION) . CloseExpression;
OPERATION_EXPRESSION        -> operand:(VARIABLE | LITERAL) . (operation:Arithmetic . source:EXPRESSION)?;
VARIABLE                    -> name:Word . (OpenSubscript . index:EXPRESSION . CloseSubscript)?;
LITERAL                     -> Measure | String;
